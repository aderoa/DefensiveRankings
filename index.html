#!/usr/bin/env python3 """ nba_defense_history.py — backfill the Defensive Rating index with every season since 2015-16 (first year of hustle stats), using the SAME fetch + scoring as the live pipeline. It reuses your existing code so historical DEF_SCOREs match the current-season logic exactly (no formula drift): fetch_defense_tabs <- nba_defense_stats.py compute_defensive_rankings <- nba_season_stats.py For each season it runs fetch -> compute -> the same DEF_SCORE table your current-season 'season' section produces, then emits it as a [YYYY-YY] section block. OUTPUT: defense_history_sections.csv — one block per season: [2015-16]
[2016-17] ... Append that to the `Defensive_Rankings` sheet AFTER the existing sections. The index already parses [MARKER] blocks by name, so column order doesn't matter — only the headers do. RUN WHERE stats.nba.com IS REACHABLE: Google Colab is easiest (its network reaches stats.nba.com and bypasses the local antivirus HTTPS inspection that blocks it on your machine). Upload these three files alongside it — nba_defense_history.py, nba_defense_stats.py, nba_season_stats.py — plus `!pip install nba_api pandas`, then run and download defense_history_sections.csv. python nba_defense_history.py # all seasons 2015-16 .. 2024-25 python nba_defense_history.py 2015-16 # just one season (re-run a failure) """ import csv import sys import time import traceback import pandas as pd try: from nba_defense_stats import fetch_defense_tabs from nba_season_stats import compute_defensive_rankings except Exception as e: # noqa: BLE001 print("Could not import the pipeline functions. Make sure nba_defense_stats.py AND " "nba_season_stats.py are in the same folder as this script.") print("Import error:", e) raise # 2016-17 .. 2025-26 (2015-16 excluded: broken first-year hustle data). (2025-26 is already the live 'season' section.) SEASONS = [f"{y}-{str(y + 1)[-2:]}" for y in range(2016, 2026)] OUT = "defense_history_sections.csv" def season_ranking(season): """Fetch + score one season, returning the DEF_SCORE DataFrame (or None).""" data = fetch_defense_tabs(season, "Regular Season") if not data: return None result = compute_defensive_rankings(data, period="season") # same call main() uses if result is None: return None df = result[0] if isinstance(result, tuple) else result # returns (df, ref_stats) return df if (df is not None and len(df) > 0) else None def main(): only = sys.argv[1] if len(sys.argv) > 1 else None seasons = [only] if only else SEASONS blocks = [] for s in seasons: print(f"[{s}] fetch + score ...", flush=True) try: df = season_ranking(s) except Exception as e: # noqa: BLE001 print(f" !! {s} FAILED: {str(e)[:120]} (re-run one season: python {sys.argv[0]} {s})") traceback.print_exc() continue if df is None: print(f" ⚠ {s}: no data returned") continue top = df.iloc[0] print(f" ✓ {s}: {len(df)} players — top: {top.get('PLAYER', '?')} ({top.get('DEF_SCORE', '?')})") blocks.append((s, df)) time.sleep(2) # be polite to stats.nba.com between seasons if not blocks: print("\nNothing produced — stats.nba.com is unreachable from here. Run this in Google Colab.") sys.exit(1) cols = list(blocks[0][1].columns) # header names the index maps by with open(OUT, "w", encoding="utf-8", newline="") as f: w = csv.writer(f) for s, df in blocks: w.writerow([f"[{s}]"]) # section marker w.writerow(cols) # header row for _, row in df[cols].iterrows(): w.writerow(["" if pd.isna(v) else v for v in row.tolist()]) print(f"\nWrote {len(blocks)} season block(s) -> {OUT}") print("Next: append it to the Defensive_Rankings sheet, and add [YYYY-YY] -> period entries") print("to SECTION_MAP in the index (I can do that HTML part for you).") if __name__ == "__main__": main()
